# Import required libraries
# pandas → for data loading and manipulation
# StandardScaler → for normalization of sensor values
# numpy → for numerical operations
import pandas as pd
from sklearn.preprocessing import StandardScaler
import numpy as np

# Load the NASA CMAPSS dataset
# sep=r"\s+" → dataset is space separated
# header=None → dataset does not contain column names
df = pd.read_csv("train_FD001.txt", sep=r"\s+", header=None)

# Define column names:
# unit → engine ID
# cycle → time step of engine operation
# op1, op2, op3 → operational settings
# sensor1–sensor21 → sensor measurements
columns = ['unit', 'cycle',
           'op1', 'op2', 'op3'] + \
          [f'sensor{i}' for i in range(1, 22)]

df.columns = columns

# Display first 20 rows to understand the structure of the dataset
df.head(20)

# Count number of unique engines in dataset
print("Number of engines:", df['unit'].nunique())

# Find maximum cycle (failure point) for each engine
# This represents the total life of each engine
max_cycles = df.groupby('unit')['cycle'].max()

print("First 10 engines max cycles:")
print(max_cycles.head(10))

# Identify all sensor columns
sensor_cols = [col for col in df.columns if 'sensor' in col]

# Print total number of sensors (should be 21)
print("Number of sensors:", len(sensor_cols))

# Check for missing values in each column
# CMAPSS dataset is usually clean (no missing values)
print("Missing values per column:")
print(df.isnull().sum())

# ---------------------------
# RUL CALCULATION
# ---------------------------
# RUL (Remaining Useful Life) = max_cycle_of_engine – current_cycle
# This becomes the target variable for regression model
df['RUL'] = df['unit'].map(max_cycles) - df['cycle']

# Display first 5 rows to verify RUL calculation
print(df.head(5))

# ---------------------------
# NORMALIZATION OF SENSOR VALUES
# ---------------------------
# We normalize sensor values so that all features are on similar scale
# This helps machine learning models learn better
sensor_cols = [col for col in df.columns if 'sensor' in col]

scaler = StandardScaler()

# Fit scaler on sensor columns and transform them
df[sensor_cols] = scaler.fit_transform(df[sensor_cols])

# Display first 5 rows after normalization
print(df.head(5))

# ---------------------------
# FEATURE SELECTION FOR MODEL
# ---------------------------
# We use only sensor values as input features
# unit and RUL are not used as input features
feature_cols = sensor_cols

# ---------------------------
# FUNCTION TO CREATE ROLLING WINDOW SEQUENCES
# ---------------------------
# seq_length = number of time steps (window size)
# For each engine:
#   Take sequences of 30 cycles → predict RUL at last cycle of that window
def create_sequences(df, seq_length, feature_cols):
    X = []  # input sequences
    y = []  # target RUL values

    # Loop through each engine separately
    for unit in df['unit'].unique():
        unit_df = df[df['unit'] == unit]

        # Extract sensor values and RUL for this engine
        data = unit_df[feature_cols].values
        rul = unit_df['RUL'].values

        # Create rolling windows
        for i in range(len(unit_df) - seq_length):
            # Take 30 time steps as one sample
            X.append(data[i:i+seq_length])

            # Target is RUL at the last time step of the window
            y.append(rul[i+seq_length])

    return np.array(X), np.array(y)

# Define window size
SEQ_LENGTH = 30

# Generate sequences
X, y = create_sequences(df, SEQ_LENGTH, feature_cols)

# Print shapes:
# X → (samples, time_steps, features)
# y → (samples,)
print("X shape:", X.shape)
print("y shape:", y.shape)

# ---------------------------
# FLATTEN SEQUENCES FOR RANDOM FOREST
# ---------------------------
# Random Forest does not accept 3D input
# So we convert (samples, 30, 21) → (samples, 630)
X_flat = X.reshape(X.shape[0], -1)
print("Flattened X shape:", X_flat.shape)

# ---------------------------
# TRAIN-TEST SPLIT
# ---------------------------
# Split dataset into 80% training and 20% testing
from sklearn.model_selection import train_test_split

X_train, X_test, y_train, y_test = train_test_split(
    X_flat, y, test_size=0.2, random_state=42
)

# ---------------------------
# BASELINE MODEL – RANDOM FOREST REGRESSOR
# ---------------------------
# Random Forest is used as a simple baseline model for RUL prediction
from sklearn.ensemble import RandomForestRegressor

model = RandomForestRegressor(n_estimators=100, random_state=42)

# Train the model on training data
model.fit(X_train, y_train)

# Predict RUL on test data
y_pred = model.predict(X_test)

# ---------------------------
# MODEL EVALUATION – RMSE
# ---------------------------
# RMSE (Root Mean Squared Error) measures prediction error
# Lower RMSE = better model performance
from sklearn.metrics import mean_squared_error

rmse = np.sqrt(mean_squared_error(y_test, y_pred))
print("RMSE:", rmse)
